import React from 'react';
import ReactDOM from 'react-dom/client';

// Small, focused components
interface ProgressBarProps {
  percentage: number;
  color?: string;
}

function ProgressBar({ percentage, color = '#22c55e' }: ProgressBarProps): JSX.Element {
  return (
    <div className="progress-bar">
      <div 
        className="progress-fill"
        style={{
          width: `${percentage}%`,
          backgroundColor: color
        }}
      />
    </div>
  );
}

interface HabitIconProps {
  icon: string;
  completed: boolean;
}

function HabitIcon({ icon, completed }: HabitIconProps): JSX.Element {
  return (
    <div className={`habit-icon ${completed ? 'completed' : ''}`}>
      {icon}
    </div>
  );
}

interface CompletionButtonProps {
  completed: boolean;
  onClick: () => void;
}

function CompletionButton({ completed, onClick }: CompletionButtonProps): JSX.Element {
  return (
    <button 
      className={`completion-btn ${completed ? 'completed' : ''}`}
      onClick={onClick}
    >
      {completed ? 'âœ“' : 'â—‹'}
    </button>
  );
}

// Compose smaller components into larger ones
interface HabitItemProps {
  habit: {
    id: number;
    name: string;
    icon: string;
    completedDays: number;
    targetDays: number;
    completedToday: boolean;
  };
  onToggle: (id: number) => void;
}

function HabitItem({ habit, onToggle }: HabitItemProps): JSX.Element {
  const percentage = (habit.completedDays / habit.targetDays) * 100;
  
  return (
    <div className="habit-item">
      <div className="habit-header">
        <HabitIcon icon={habit.icon} completed={habit.completedToday} />
        <div className="habit-info">
          <h3>{habit.name}</h3>
          <p>{habit.completedDays} / {habit.targetDays} days</p>
        </div>
        <CompletionButton 
          completed={habit.completedToday}
          onClick={() => onToggle(habit.id)}
        />
      </div>
      <ProgressBar percentage={percentage} />
    </div>
  );
}

// Even larger composition
function HabitList(): JSX.Element {
  const [habits, setHabits] = React.useState([
    { id: 1, name: 'Morning Meditation', icon: 'ðŸ§˜', completedDays: 15, targetDays: 30, completedToday: true },
    { id: 2, name: 'Read for 30 mins', icon: 'ðŸ“š', completedDays: 10, targetDays: 30, completedToday: false },
    { id: 3, name: 'Exercise', icon: 'ðŸƒ', completedDays: 20, targetDays: 30, completedToday: true },
    { id: 4, name: 'Drink Water', icon: 'ðŸ’§', completedDays: 25, targetDays: 30, completedToday: false }
  ]);

  const handleToggle = (id: number) => {
    setHabits(habits.map(habit => 
      habit.id === id 
        ? { ...habit, completedToday: !habit.completedToday }
        : habit
    ));
  };

  const completedToday = habits.filter(h => h.completedToday).length;

  return (
    <div className="habit-list">
      <div className="summary">
        <h2>Today's Progress</h2>
        <p>{completedToday} of {habits.length} habits completed</p>
      </div>
      <div className="habits">
        {habits.map(habit => (
          <HabitItem 
            key={habit.id}
            habit={habit} 
            onToggle={handleToggle}
          />
        ))}
      </div>
    </div>
  );
}

function App(): JSX.Element {
  return <HabitList />;
}

// Render the app
const root = ReactDOM.createRoot(document.getElementById('root')!);
root.render(<App />);
