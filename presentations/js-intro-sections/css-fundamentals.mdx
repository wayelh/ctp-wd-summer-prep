h1 {
    color: #333;
    font-family: Arial, sans-serif;

    background-color: yellow;
    padding: 5px;
}

/* ID Selector */

    font-weight: bold;
}

/* Descendant Selector */
.container p {
    color: blue;
}

/* Multiple Classes */
li.active {
    background-color: #4CAF50;
    color: white;
    padding: 5px;
}

/* Pseudo-class */
button:hover {
    cursor: pointer;
    opacity: 0.8;
}

/* Attribute Selector */
button.primary {

    color: white;

describe('Habit Card Styling', () => {
    const card = document.querySelector('.habit-card');
    const styles = window.getComputedStyle(card);
    
    it('should have rounded corners', () => {
        expect(styles.borderRadius).to.not.equal('0px');
    });
    
    it('should have a shadow', () => {
        expect(styles.boxShadow).to.not.equal('none');
    });
    
    it('should use flexbox or grid', () => {
        const display = styles.display;
        expect(['flex', 'grid'].includes(display)).to.be.true;
    });
